##二叉排序树的实现和查找

---------
#include<stdio.h>
#include<stdlib.h>
#include<iostream>
using namespace std;
typedef struct node
{
	int key;
	int data;
	struct node *lchild,*rchild;
}bstnode;
int sum=0,flag;
int insert(bstnode *&b,int k)
{
	if(b==NULL)
	{
		b=(bstnode *)malloc(sizeof(bstnode));
		b->key=k;
		b->lchild=NULL;
		b->rchild=NULL;
		return 1;
	}
	else if(k==b->key)
		return 0;
	else if(k>b->key)
		return insert(b->rchild,k);
	else 
		return insert(b->lchild,k);
	return 0;
}
bstnode *search(bstnode *&b,int k)
{
	sum++;
	if(b==NULL)
		return b;
	else if(b->key==k)
	{
		flag=1;
		return b;
	}
	else if(k>b->key)
		return search(b->rchild,k);
	else
		return search(b->lchild,k);
}
int main()
{
	bstnode *b=NULL;//记住，一定是b=NULL
	int n,m,k;
	cin>>n;
	while(n--)
	{
		cin>>m;
		insert(b,m);
	}
	cin>>k;
	search(b,k);
	if(flag)
		cout<<sum;
	else
		cout<<"-1";
	return 0;
}
